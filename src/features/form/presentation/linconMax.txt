const NumberFormatCustom = React.forwardRef(function NumberFormatCustom(
	props,
	ref,
) {
	const { onChange, ...other } = props;

	return (
		<NumberFormat
			{...other}
			getInputRef={ref}
			onValueChange={values => {
				onChange({
					target: {
						name: props.name,
						value: values.value,
					},
				});
			}}
			decimalSeparator=","
			allowNegative={false}
			decimalScale={2}
			thousandSeparator={false}
			isNumericString
		/>
	);
});
